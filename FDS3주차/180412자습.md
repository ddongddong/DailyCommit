### String 문제
* 4번문제
```js
function countChar(str) {
  const obj = {};
  for (let i = 0; i < str.length; i++) {
    const char = str[i];
    // 만약 속성이 없다면
    if (obj[char] == null) { // null check
      // 속성 값에 1 저장
      obj[char] = 1;
    } else {
      // 속성이 있으면 1 증가
      obj[char]++;
    }
  }
  return obj;
}
```
